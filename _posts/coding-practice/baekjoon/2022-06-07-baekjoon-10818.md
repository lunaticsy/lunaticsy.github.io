import kotlin.collections.ArrayList

fun q10818() = with(Scanner(System.`in`)) {
    val list = ArrayList<Int>()
    for(i in 0 until nextInt())
        list.add(nextInt())

    println(list.minOf { it }.toString() + " " + list.maxOf { it }.toString())
}

import java.util.*

fun main(args: Array<String>) {
    q10952();
}

fun q10952() = with(Scanner(System.`in`)) {
    val inputs = LinkedList<Int>()
    while (true) {
        val first = nextInt()
        val second = nextInt()
        if(first == 0 && second == 0) break;
        inputs.add(first + second);
    }
    inputs.forEach {println(it)}
}
